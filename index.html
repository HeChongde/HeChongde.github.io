基于区块链的简单投票（windows系统）

本教程旨在实现一个简单的具备前端页面、可以操作基于区块链的投票的程序。希望以此打开区块链程序开发的大门。

本文仅说明操作，及注意事项，默认操作者已经具备以下能力：

  • 理解区块链和以太坊的部分基础知识。
  • 掌握solidity基本语法

教程大纲：

  1. 系统成品了解
  2. Remix编译、发布、调试。
  3. 账户与钱包
  4. Gath客户端及其简单操作。
  5. web3js前端编写（前端连接后端的接口）
  6. 智能合约发布到区块链
  7. 项目完成，进行项目运行。

简单介绍：

本文会首先展示系统成品，便于大家了解。然后介绍solidity语言编写的智能合约在Remix编译器中的编写，调试，发布。区块链上进行的操作大多需要有一定的花费（gas）所以需要向大家介绍账户和钱包的使用和操作，这在开发中也是必不可少的。然后是Gath客户端的操作，它是用go语言编写的一个以太坊客户端，这里主要用它来进行私有链的创建、配置、挖矿等操作。了解了这些之后就可以把我们编写的投票智能合约发布到我们部署好的私有链上了。有了这些之后实际上我们的底层区块链程序已经完成了。最后就是做一个简单的前端页面进行调用，完成简单操作。

//版本
pragma solidity ^0.4.0;
//投票合约
contract Ballot{
    //投票者结构体
    struct Voter {
        uint weight;   //投票权重
        bool voted;    //是否已投过票
        uint vote;     //若已经投票的投票编号
        address delegate;  //投票权委托别人的地址
    }
    //提案结构体
    struct Proposal {
        bytes32 name; //姓名
        uint voteCount;  //获得票数
    }
    
    address chairperson;   //主席
    
    mapping(address => Voter) voters;  //映射
    
    Proposal[] proposals;  //提案列表
    
    constructor() public { //构造函数
        chairperson = msg.sender;   //设置主席为合约发布者
        voters[chairperson].weight = 1;  //主席的投票权重设为1
    }
    //////////////////////////////////////////////
    function setProposals(bytes32[] proposalNames) public {//设置提案名单
        require((msg.sender == chairperson));  //仅主席可以执行此操作
         for(uint j = 0; j<proposalNames.length; j++){
            proposals.push(Proposal({  //添加提案
                name:proposalNames[j],
                voteCount: 0
            }));
        }
    }
    
    /////////////////////////////////////////////////////////////////////////////////////////
     function giveRightToVote(address voter) public {  //给一个地址授予投票权
        require((msg.sender == chairperson) && !voters[voter].voted && (voters[voter].weight == 0));//只有主席可执行此操作，该地址未投过票，投票权重为0
        voters[voter].weight = 1;  //将其投票权重设为1
    }
    
    function giveRightToVoteByBatch(address[] batch) public { //给一批地址授予投票权
        require(msg.sender == chairperson);  //仅主席
        
        for(uint i = 0;i < batch.length; i++){ //循环单个添加
            address voter = batch[i];
            require(!voters[voter].voted && (voters[voter].weight == 0));
            voters[voter].weight = 1;
        }
    }
    function delegate(address to) public payable {  //委托别人投票（投票者的操作）
        Voter storage sender = voters[msg.sender];  
        require((!sender.voted) && (sender.weight != 0));//发起者未投过票且权重不为0
        
        require(to != msg.sender); //不能委托自己
        
        while(voters[to].delegate != address(0)) {  //防止被委托者又委托别人，所以找到最后一个人
            to = voters[to].delegate;
            require(to != msg.sender);
        }
        
        sender.voted = true;//设置发起者已经投过票
        sender.delegate = to;  //委托人是to
        Voter storage _delegate = voters[to];
        if(_delegate.voted){//如果委托人已经投过票，则这一票投给和委托人相同的提案
            proposals[_delegate.vote].voteCount += sender.weight;
        }else {//如果委托人还未投过票，则将该权重添加到委托人身上
            _delegate.weight += sender.weight;
        }
    }
    
    function vote(uint proposal) public { //投票函数  输入提案编号进行投票
        require(proposal < proposals.length);  //提案编号不得越界
        
        Voter storage sender = voters[msg.sender];
        require((!sender.voted) && (sender.weight != 0));
        sender.voted = true;
        sender.vote = proposal;
        
        proposals[proposal].voteCount += sender.weight;//票数累加到被投票提案
    }
    
    function winningProposal() public view returns(uint[] winningProposals) {//获取赢得投票的提案名单
        uint[] memory tempWinner = new uint[](proposals.length);
        uint winningCount = 0;
        
        uint winningVoteCount = 0;
        
        for(uint p = 0; p < proposals.length;p++){
            if(proposals[p].voteCount > winningVoteCount){
                winningVoteCount = proposals[p].voteCount;
                
                tempWinner[0] = p;
                winningCount = 1;
            }else if(proposals[p].voteCount == winningVoteCount){
                tempWinner[winningCount] = p;
                winningCount++;
            }
        }
        
        winningProposals = new uint[](winningCount);
        
        for(uint q = 0; q < winningCount; q++){
            winningProposals[q] = tempWinner[q];
        }
        
        return winningProposals;
    }
    
    
    function winnerName() public view returns(bytes32[] winnerNames) {//获取赢得提案的名字（可能多个）
        uint[] memory winningProposals = winningProposal();
        
        winnerNames = new bytes32[](winningProposals.length);
        
        for(uint p = 0; p < winningProposals.length; p++){
            winnerNames[p] = proposals[winningProposals[p]].name;
        }
        return winnerNames;
    }
   //////////////////////////////////////////////////////////////////////////////////
    //通过姓名获取到其票数
    function getNumByName(bytes32 name) public view returns(uint num){
         for(uint p = 0; p < proposals.length; p++){
           if(proposals[p].name == name){
               return proposals[p].voteCount;
           }
        }
    }
    
    //获取所有提案的姓名
    function allName()public view returns(bytes32[] name){
        bytes32[] memory names = new bytes32[](proposals.length);
          for(uint p = 0; p < proposals.length;p++){
           names[p] = proposals[p].name;
        }
        return names;
    } 
}

注：

  • 代码中有主席这一项，目前被设置为合约的发布者，其实也可以把其他地址设置为主席，具有很大灵活性
  • 设置提案名单的函数是我自己添加的函数，目的在于在前端输入设置名单，因为remix编译默认只能设置0x1233411这种形式的字符。在实现的系统中，它充当初始化名单的作用，不能设置第二次名单。
  • 为求简便，前端实现中默认只设置5个候选名单
  • 该投票合约考虑比较周全，比较完善，由于本系统仅作演示，前端实现仅调用其中一两个函数，感兴趣者可以自行扩展。

第二步，geth 基本操作

  1. 首先要进入geth内部的控制台，我们来看一句长的命令，以后可以直接使用。

    xxxxxxxxxx
    geth --datadir test --networkid 666 --rpc --rpccorsdomain * --allow-insecure-unlock --nodiscover console
    • geth代表启动geth
    • --datadir代表启动的目录，要和我们创世区块目录一致。
    • --netwokid 123 代表私有链的id，注意，这和创世区块中的chainId本身不一样，但请设置为同样的，因为在设置的不一样时，我们遇到了莫名其妙的问题。
    • --rpc是启动rpc接口，这对我们是一个十分有用的东西，只有这样才能暴露出接口供前端页面的调用和钱包链接。
    • --rpccorsdomain * 代表跨域访问，为了让我们的Remix及其他需要的页面也能访问到，所以要添加该属性。
    • --allow-insecure-unlock 账户转账等操作需要解锁操作，而目前新版本为了安全默认不可解锁，添加此属性则可进行解锁。
    • console 代表启动控制台。只有添加了这句，才可以启动geth控制台进行命令行操作。

    这就进入控制台了。

  2. 然后是几个我们用到的简单命令

    • eth.accounts 查询账户列表
    • personal.newAccount() 创建新账户（需要输入密码）
    • personal.unlockAccount(eth.accounts[0]) 解锁账户，其中，eth.accounts[0]得到的是账户列表中的第一个账户地址，也就是我们默认在使用的地址。
    • eth.getBalance(eth.accounts[0]) 查询余额
    • miner.start() 启动挖矿 十分重要的操作，这是获得资金和区块同步，信息执行的重要操作
    • miner.stop() 结束挖矿

    其余还有转账等操作命令，大家可选择性自行了解。

智能合约部署到区块链

好了，现在，我们只需要把合约部署到私有链上，然后进行获得合约地址，就可以了。

首先打开已经连接到私有链的MetaMask和Gath控制台（确保账户有资金，MetaMask连接到私有链）

然后进入到Remix编译器，编译成功，并进入发布的页面（忘了的童鞋请移步前文哦）

这次，我们不在javascript VM中部署，而选择injected Web3

Metamask会弹出来，让你选择连接的账户：选择你之前连接的私有链账户即可

此时，remix也连接成功了：（如果没有连接成功，在Metamask连接正常的情况下，尝试刷新Remix页面，可能是Remix和MetaMask连接没有同步）

点击发布，MetaMask会弹出来，显示花费的ges数目，默认即可点击确认

然后MetaMask窗口消失，如果手动打开，会发现显示合约在等待部署。

熟悉区块链知识的童鞋应该已经知道了，此时系统正在等待被挖矿，只有挖出新的区块，才可以将部署合约的信息同步到区块链上。

所以在Geth中启动挖抗，挖出块后会提示部署成功，停止挖矿。

好的，这就把我们的合约部署到了区块链上了，可以看到Remix下方出现部署成功的合约：

注意事项：

  • 在部署时我们遇到了莫名其妙的问题：部署失败，或者一直在padding等待部署。而且在重装MetaMask之后可能恢复正常或者反反复复，时好时坏。解决方法是：把创世区块chainId和networkid设置为一样的值。
  • 如果还不行，可以尝试修改gaslimt的值。
